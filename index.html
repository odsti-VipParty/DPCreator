    <script>

        const imageInput = document.getElementById('imageInput');

        const uploadedImage = document.getElementById('uploadedImage');

        const frame = document.getElementById('frame');

        const canvasDiv = document.getElementById('canvas');

        const downloadBtn = document.getElementById('downloadBtn');


        let offsetX = 0, offsetY = 0, isDragging = false;


        imageInput.addEventListener('change', function(event) {

            const file = event.target.files[0];

            if (file) {

                const reader = new FileReader();

                reader.onload = function(e) {

                    uploadedImage.src = e.target.result;

                    uploadedImage.style.position = 'absolute';

                };

                reader.readAsDataURL(file);

            }

        });


        uploadedImage.addEventListener('mousedown', function(event) {

            isDragging = true;

            offsetX = event.clientX - uploadedImage.offsetLeft;

            offsetY = event.clientY - uploadedImage.offsetTop;

        });


        document.addEventListener('mousemove', function(event) {

            if (isDragging) {

                let x = event.clientX - offsetX;

                let y = event.clientY - offsetY;

                uploadedImage.style.left = x + 'px';

                uploadedImage.style.top = y + 'px';

            }

        });


        document.addEventListener('mouseup', function() {

            isDragging = false;

        });


        downloadBtn.addEventListener('click', function() {

            const canvas = document.createElement('canvas');

            const ctx = canvas.getContext('2d');


            canvas.width = 300;

            canvas.height = 300;


            ctx.drawImage(uploadedImage, 0, 0, 300, 300);

            ctx.drawImage(frame, 0, 0, 300, 300);


            const link = document.createElement('a');

            link.download = 'VIP_DP.png';

            link.href = canvas.toDataURL('image/png');

            link.click();

        });

    </script>
